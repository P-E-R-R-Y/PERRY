165 void drawPoint(const sf::Vector3f& point) {
166
168 sf::Vector3f relativePosition = point - _camera->_position;
169
170 std::cout << "relativePosition: " << relativePosition.x << " " << relativePosition.y << " " << relativePosition.z << std::endl;
171
173 sf::Vector3f relativeRotatedPosition = _camera->_quaternion.conjugate().rotate(relativePosition);
174
175 if (relativeRotatedPosition.z >= 0) {
176 return; //skip drawing points that are behind the camera
177 }
178
179 //Don't need to move the object back to its original position because the camera is the origin for perspective projection
180 sf::Vector3f rotatedPosition = relativeRotatedPosition;
181 std::cout << "points: " << rotatedPosition.x << " " << rotatedPosition.y << " " << rotatedPosition.z << std::endl;
183 sf::Vector2f screen = _window.getView().getSize();
184
185 Matrice<float> projection = perspectiveProjection<float>(_camera->_fov, screen.x / screen.y);
186 Matrice<float> vtom = Matrice<float>({{rotatedPosition.x}, {rotatedPosition.y}, {rotatedPosition.z}, {1.f}});
187 Matrice<float> res = projection * vtom;
189 float x_ndc = res(0, 0) / res(3, 0);
190 float y_ndc = res(1, 0) / res(3, 0);
191
192 if (x_ndc < -1 || x_ndc > 1 || y_ndc < -1 || y_ndc > 1) {
193 return; //skip drawing points that are outside the screen
194 }
195
196 //NDC to screen coordinates
197 float screenX = (x_ndc + 1.0f) * 0.5f * screen.x;
198 float screenY = (1.0f - y_ndc) * 0.5f * screen.y;
199
200 // Draw the point
201 sf::CircleShape circle(1);
202 circle.setPosition(screenX, screenY);
203 circle.setFillColor(sf::Color::White);
204 circle.setOutlineColor(sf::Color::White);
205 circle.setOutlineThickness(1);
206 _window.draw(circle);
207
208 return;
209 }
