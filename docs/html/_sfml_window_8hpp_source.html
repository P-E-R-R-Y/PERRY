101 void drawPoint(const sf::Vector3f& point) {
102
104 sf::Vector3f relativePosition = point - _camera->_position;
105
106 std::cout << "relativePosition: " << relativePosition.x << " " << relativePosition.y << " " << relativePosition.z << std::endl;
107
109 sf::Vector3f relativeRotatedPosition = _camera->_quaternion.conjugate().rotate(relativePosition);
110
111 if (relativeRotatedPosition.z >= 0) {
112 return; //skip drawing points that are behind the camera
113 }
114
115 //Don't need to move the object back to its original position because the camera is the origin for perspective projection
116 sf::Vector3f rotatedPosition = relativeRotatedPosition;
117 std::cout << "points: " << rotatedPosition.x << " " << rotatedPosition.y << " " << rotatedPosition.z << std::endl;
119 sf::Vector2f screen = _window.getView().getSize();
120
121 Matrice<float> projection = perspectiveProjection<float>(_camera->_fov, screen.x / screen.y);
122 Matrice<float> vtom = Matrice<float>({{rotatedPosition.x}, {rotatedPosition.y}, {rotatedPosition.z}, {1.f}});
123 Matrice<float> res = projection * vtom;
125 float x_ndc = res(0, 0) / res(3, 0);
126 float y_ndc = res(1, 0) / res(3, 0);
127
128 if (x_ndc < -1 || x_ndc > 1 || y_ndc < -1 || y_ndc > 1) {
129 return; //skip drawing points that are outside the screen
130 }
131
132 //NDC to screen coordinates
133 float screenX = (x_ndc + 1.0f) * 0.5f * screen.x;
134 float screenY = (1.0f - y_ndc) * 0.5f * screen.y;
135
136 // Draw the point
137 sf::CircleShape circle(1);
138 circle.setPosition(screenX, screenY);
139 circle.setFillColor(sf::Color::White);
140 circle.setOutlineColor(sf::Color::White);
141 circle.setOutlineThickness(1);
142 _window.draw(circle);
143
144 return;
145 }
