172 void drawPoint(const sf::Vector3f& point) {
173
175 sf::Vector3f relativePosition = point - _camera->_position;
176
177 //std::cout << "relativePosition: " << relativePosition.x << " " << relativePosition.y << " " << relativePosition.z << std::endl;
178
180 sf::Vector3f relativeRotatedPosition = _camera->_quaternion.conjugate().rotate(relativePosition);
181
182 if (relativeRotatedPosition.z >= 0) {
183 return; //skip drawing points that are behind the camera
184 }
185
186 //Don't need to move the object back to its original position because the camera is the origin for perspective projection
187 sf::Vector3f rotatedPosition = relativeRotatedPosition;
188 //std::cout << "points: " << rotatedPosition.x << " " << rotatedPosition.y << " " << rotatedPosition.z << std::endl;
190 sf::Vector2f screen = _window.getView().getSize();
191
192 Matrice<float> projection = perspectiveProjection<float>(_camera->_fov, screen.x / screen.y);
193 Matrice<float> vtom = Matrice<float>({{rotatedPosition.x}, {rotatedPosition.y}, {rotatedPosition.z}, {1.f}});
194 Matrice<float> res = projection * vtom;
196 float x_ndc = res(0, 0) / res(3, 0);
197 float y_ndc = res(1, 0) / res(3, 0);
198
199 if (x_ndc < -1 || x_ndc > 1 || y_ndc < -1 || y_ndc > 1) {
200 return; //skip drawing points that are outside the screen
201 }
202
203 //NDC to screen coordinates
204 float screenX = (x_ndc + 1.0f) * 0.5f * screen.x;
205 float screenY = (1.0f - y_ndc) * 0.5f * screen.y;
206
207 // Draw the point
208 sf::CircleShape circle(1);
209 circle.setPosition(screenX, screenY);
210 circle.setFillColor(sf::Color::White);
211 circle.setOutlineColor(sf::Color::White);
212 circle.setOutlineThickness(1);
213 _window.draw(circle);
214
215 return;
216 }

274void SfmlWindow::drawPoly(graphic::IPolygon *polygon) {
275 SfmlPolygon *sfmlPolygon = static_cast<SfmlPolygon *>(polygon);
276 std::vector<graphic::triangle_t> triangles = sfmlPolygon->_triangles;
277
278 for (int i = 0; i < triangles.size(); i++) {
279 sf::ConvexShape cpolygon;
280 cpolygon.setPointCount(3);
281 cpolygon.setPoint(0, sf::Vector2f(triangles[i].A.x, triangles[i].A.y));
282 cpolygon.setPoint(1, sf::Vector2f(triangles[i].B.x, triangles[i].B.y));
283 cpolygon.setPoint(2, sf::Vector2f(triangles[i].C.x, triangles[i].C.y));
284 cpolygon.setFillColor(sfmlPolygon->_color);
285
286 cpolygon.setPosition(0, 0);
287 _window.draw(cpolygon);
288 }
