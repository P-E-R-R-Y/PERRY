182template <typename T>
183Matrice<T> perspectiveProjection(T fov, T aspectRatio) {
184 // Convert FOV from degrees to radians
185 T fovRad = fov * M_PI / 180.0;
186 T nearPlane = 0.001;
187 T farPlane = 1000.0;
188
189 // Create the perspective projection matrix
190 std::vector<std::vector<T>> proj = {{
191 {1.0f / (aspectRatio * tan(fovRad / 2.f)), 0.f, 0.f, 0.f},
192 {0.f, 1.f / tan(fovRad / 2.f), 0.f, 0.f},
193 {0.f, 0.f, (farPlane + nearPlane) / (nearPlane - farPlane), (2 * farPlane * nearPlane) / (nearPlane - farPlane)},
194 {0.f, 0.f, -1.f, 0.f}
195 }};
196
197 return Matrice<T>(proj);
198}

212template <typename T>
213Matrice<T> orthographicProjection(T left, T right, T bottom, T top, T nearPlane, T farPlane) {
214 // Create the orthographic projection matrix
215 std::array<std::array<T, 4>, 4> proj = {{
216 {2.0 / (right - left), 0, 0, -(right + left) / (right - left)},
217 {0, 2.0 / (top - bottom), 0, -(top + bottom) / (top - bottom)},
218 {0, 0, -2.0 / (farPlane - nearPlane), -(farPlane + nearPlane) / (farPlane - nearPlane)},
219 {0, 0, 0, 1}
220 }};
221
222 return Matrice<T>(proj);
